SET(LIB_NAME PhysicsLab)
SET(PLUGIN_HEADERS
    PhysicsLab.h
    BulletHandler.h
    AntiGravityField.h
    ObjectFactory.h
    TriangleVisitor.h
)

    "SHARED"
    ${PLUGIN_HEADERS}
    PhysicsLab.cpp
    BulletHandler.cpp
    ObjectFactory.cpp
    TriangleVisitor.cpp
)

FIND_PACKAGE(OASCLIENT)
INCLUDE_DIRECTORIES(${OASCLIENT_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${LIB_NAME} ${OASCLIENT_LIBRARY})

FIND_PACKAGE(Bullet)

TARGET_LINK_LIBRARIES(${LIB_NAME} ${BULLET_LIBRARIES})

INCLUDE_DIRECTORIES(${OSG_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${LIB_NAME} ${OSG_LIBRARIES})

INSTALL(TARGETS ${LIB_NAME} DESTINATION lib/plugins)
